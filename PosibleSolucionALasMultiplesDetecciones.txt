import os
import glob

def encontrar_archivos_coordenadas(self, ruta_directorio_coordenadas, nombre_base_imagen):
    # Esta función busca todos los archivos de coordenadas correspondientes a la imagen base.
    # Por ejemplo, frame_00000.png_1.txt, frame_00000.png_2.txt, etc.
    patron_busqueda = f"{nombre_base_imagen}_*.txt"
    ruta_busqueda = os.path.join(ruta_directorio_coordenadas, patron_busqueda)
    archivos_coordenadas = glob.glob(ruta_busqueda)
    return archivos_coordenadas
